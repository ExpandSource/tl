#10-3  if#  조건문
```bash
if [[ $NAME == "Alice" ]]; then
  echo "안녕, Alice!"
elif [[ $NAME == "Bob" ]]; then
  echo "안녕, Bob!"
else
  echo "누구세요?"
fi
```

#비교 연산자#
`==` `!=`          문자열 같음/다름
`-eq` `-ne`        숫자 같음/다름
`-gt` `-lt`        숫자 크다/작다
`-z "$VAR"`        변수가 비어있음
`-n "$VAR"`        변수가 비어있지 않음
`-f file`          파일 존재
`-d dir`           디렉토리 존재
`-x file`          실행 권한 있음

`&&`  앞이 성공하면 실행
`||`  앞이 실패하면 실행
