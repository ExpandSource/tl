#10-4  loop#  반복문
#for#
```bash
for i in 1 2 3 4 5; do
  echo "번호: $i"
done

for file in *.txt; do
  echo "처리 중: $file"
done

for ((i=0; i<10; i++)); do
  echo $i
done
```

#while#
```bash
count=0
while [[ $count -lt 5 ]]; do
  echo $count
  ((count++))
done
```

#until — 조건이 참이 될 때까지#
```bash
until ping -c1 google.com &>/dev/null; do
  echo "대기 중..."; sleep 2
done
```
