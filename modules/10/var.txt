#10-2  var#  변수 선언·사용
#변수 선언#
`NAME="Alice"`     공백 없이 붙여 씀 (공백 있으면 오류)
`echo $NAME`       변수 사용
`echo "${NAME}!"`  문자열 이어쓰기

#특수 변수#
`$0`   스크립트 이름
`$1`   첫 번째 인수 (`./script.sh arg1`)
`$#`   인수 개수
`$?`   직전 명령 종료 코드 (0=성공, 1+=실패)
`$$`   현재 스크립트 PID
`$@`   모든 인수

#산술#
`result=$((3 + 5))`
`echo $((10 / 3))`   정수 나눗셈

`readonly PI=3.14`   상수 선언
`unset NAME`         변수 삭제
*← tl p*   *10-2 / 6*   *tl n →*
