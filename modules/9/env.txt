#9-2  env#  환경변수와 $PATH
`env`                  모든 환경변수 출력
`echo $HOME`           특정 변수 확인
`export VAR=value`     변수 설정 (현재 세션)
`echo 'export VAR=value' >> ~/.zshrc`  영구 설정

#$PATH 구조#
`echo $PATH`
→ `/usr/local/bin:/usr/bin:/bin:...`
명령어 입력 시 이 경로들을 순서대로 탐색.

#PATH 추가#
`export PATH="$HOME/.local/bin:$PATH"`
~/.zshrc 에 추가하면 매번 자동 설정.

`which python3`   python3 실행 파일 위치 확인
